データバインディング

なんだか、楽しいデータバインディング。意味ははっきりとは分かってない！

DEMO1

HTML

<div id="app"> // message.value なんて感じでオブジェクトな扱いが出来ていい感じ <p>{{ message.value }}</p> </div>

Script

var app = new Vue({ el: '#app', data: { message: { value: 'Hello Vue.js!' } } })

出力結果

{{ message.value }}

DEMO2

HTML

<div id="app2"> // Vueインスタンス内で定義してるプロパティも使えるよーん <p>{{ message.value.length }}</p> <p>{{ list[2] }}</p> <p>{{ list[num] }}</p> </div>

Script

var app = new Vue({ el: '#app2', data: { message: { value: 'Hello Vue.js!' }, list: ['りんご', 'ばなな', 'いちご'], num: 1 } })

出力結果

{{ message.value.length }}

{{ list[2] }}

{{ list[num] }}

var app = new Vue({ el: '#app3', data: { message: { value: 'Hello Vue.js!' } } })

出力結果

DEMO4

HTML

<div id="app4"> // .prop みたいなよく分かんない書き方もあるのよ。よく分かんないのよ <div v-bind:text-content.prop="message.value"></div> <div v-bind:scroll-top.prop="message.value">{{ message.value }}</div> </div>

Script

var app = new Vue({ el: '#app4', data: { message: { value: 'Hello Vue.js!' } }, // mounted、なんだろころ。 mounted: function() { this.scroll = 100 } })

出力結果

{{ message.value }}

DEMO5

HTML

<div id="app5"> // v-on:click で increment関数を発火させて、簡易カウンターだよ <p>{{ count }}回クリックしたで！</p> <button v-on:click="increment">カウントを増やす</button> </div>

Script

var app = new Vue({ el: '#app5', data: { count: 0 } // メソッドの定義もこんな感じ？これっていっぱいメソッド追加したらややこしくない？いい感じにモジュール化するのかな？ methods: { increment: function() { this.count += 1 } } })

出力結果

{{ count }}回クリックしたで！
