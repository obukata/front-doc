なんだか、楽しいデータバインディング。意味ははっきりとは分かってない！

<div id="app"> // message.value なんて感じでオブジェクトな扱いが出来ていい感じ <p>{{ message.value }}</p> // Vueインスタンス内で定義してるプロパティも使えるよーん <p>{{ message.value.length }}</p> <p>{{ list[2] }}</p> <p>{{ list[num] }}</p> // {{ message }} みたいなのはDOMContentとして使う。プロパティで使う場合は v-bind:value="" で使うべし <input type="text" v-bind:value="message.value"> // .prop みたいなよく分かんない書き方もあるのよ。よく分かんないのよ <div v-bind:text-content.prop="message.value"></div> <div v-bind:scroll-top.prop="message.value">{{ message.value }}</div> // v-on:click で increment関数を発火させて、簡易カウンターだよ <p>{{ count }}回クリックしたで！</p> <button v-on:click="increment">カウントを増やす</button> </div>

Script

var app = new Vue({ el: '#app', // data内に色々定義出来ていい感じよー data: { message: { value: 'Hello Vue.js!' }, count: 0, scroll: 0, list: ['りんご', 'ばなな', 'いちご'], num: 1 }, // mounted、なんだろころ。 mounted: function() { this.scroll = 100 }, // メソッドの定義もこんな感じ？これっていっぱいメソッド追加したらややこしくない？いい感じにモジュール化するのかな？ methods: { increment: function() { this.count += 1 } } })
