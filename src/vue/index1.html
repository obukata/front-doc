Vue.jsの基本機能

Vueをとりあえず動かす手順は以下3つ

  • CDN版Vue.jsを読み込む
  • HTMLにVueを動かす要素を用意する
  • new Vue()にてVueインスタンスを生成する

DEMO1

HTML

// マウントする#app <div id="app"> // {{ 変数 }} のような形なテンプレート <p>{{ message }}</p> </div>

Script

var app = new Vue({ el: '#app', data: { message: 'Hellow Vue.js!' } })

出力結果

{{ message }}

DEMO2

HTML

<div id="app2"> // v-for="" でfor文がかけます。 item in 変数 にて、配列を格納してる変数の内容を全て吐き出す。 <ol> <li v-for="item in list">{{ item }}</li> </ol> </div>

Script

var app = new Vue({ el: '#app2', data: { list: ['りんご', 'ばなな', 'いちご'] } })

出力結果

  1. {{ item }}

DEMO3

HTML

<div id="app3"> // v-on:click はjQueryの element.on('click', handleClick())・addEventListener('click', handleClick())と同じよ <button v-on:click="handleClick">Click</button><br> </div>

Script

var app = new Vue({ el: '#app3', methods: { handleClick: function(event) { alert(event.target) } } })

出力結果


DEMO4

HTML

<div id="app4"> // ↑のinputMessageをデータバインディング出来る！ <p>{{ inputMessage }}</p> // v-model:number にて数値を取得出来る…らしい <input type="text" v-model:number="inputMessage"> </div>

Script

var app = new Vue({ el: '#app4', data: { inputMessage: 'test', } })

出力結果

{{ inputMessage }}

DEMO5

HTML

<div id="app5"> // transition ってタグつけると v-enter、v-leave ってclassがつくみたい！ <transition> // v-if はif文かけるよ！ <p v-if="show">v-if show!</p> </transition> // clickするたびに true と false を切り替えるよ。↑の表示非表示を変えるよ <button v-on:click="show=!show">切り替え</button> </div>

Script

var app = new Vue({ el: '#app5', data: { show: true } })

Style

.v-enter-active, .v-leave-active { transition: opacity 1000ms; } .v-enter, .v-leave-to { opacity: 0; }

出力結果

v-if show!
